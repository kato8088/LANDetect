#include <Windows.h>

#define WM_TRAYICON (WM_USER + 1)

NOTIFYICONDATA nid;

LRESULT CALLBACK WindowProc(HWND hwnd, UINT uMsg, WPARAM wParam, LPARAM lParam) {
    switch (uMsg) {
        case WM_TRAYICON:
            if (LOWORD(lParam) == WM_LBUTTONUP) {
                // Lidar com o clique esquerdo no ícone da barra de notificações aqui
                MessageBox(NULL, L"Ícone da barra de notificações clicado!", L"Click Detected", MB_ICONINFORMATION);
            }
            break;
        default:
            return DefWindowProc(hwnd, uMsg, wParam, lParam);
    }
    return 0;
}

int main() {
    // Configurar a estrutura NOTIFYICONDATA
    nid.cbSize = sizeof(nid);
    nid.hWnd = NULL; // Preencher com o HWND da janela principal, se necessário
    nid.uID = 1;
    nid.uFlags = NIF_ICON | NIF_MESSAGE;
    nid.uCallbackMessage = WM_TRAYICON;
    nid.hIcon = LoadIcon(NULL, IDI_APPLICATION);

    // Adicionar o ícone à bandeja do sistema
    Shell_NotifyIcon(NIM_ADD, &nid);

    // Criar uma janela oculta para receber mensagens de notificação
    WNDCLASS wc = {};
    wc.lpfnWndProc = WindowProc;
    wc.hInstance = GetModuleHandle(NULL);
    wc.lpszClassName = L"TrayWindowClass";
    RegisterClass(&wc);

    HWND hwnd = CreateWindow(wc.lpszClassName, L"TrayWindow", 0, 0, 0, 0, 0, NULL, NULL, wc.hInstance, NULL);

    MSG msg;
    while (GetMessage(&msg, NULL, 0, 0)) {
        TranslateMessage(&msg);
        DispatchMessage(&msg);
    }

    // Remover o ícone da bandeja do sistema ao sair
    Shell_NotifyIcon(NIM_DELETE, &nid);

    return 0;
}

#include <windows.h>

// ID para o ícone na barra de notificações
#define ID_TRAY_ICON 1

// Procedimento de janela
LRESULT CALLBACK WindowProc(HWND hwnd, UINT uMsg, WPARAM wParam, LPARAM lParam)
{
    switch (uMsg)
    {
    case WM_CREATE:
    {
        // Criar o ícone na barra de notificações
        NOTIFYICONDATA nid = { 0 };
        nid.cbSize = sizeof(NOTIFYICONDATA);
        nid.hWnd = hwnd;
        nid.uID = ID_TRAY_ICON;
        nid.uFlags = NIF_ICON | NIF_TIP | NIF_MESSAGE;
        nid.uCallbackMessage = WM_USER + 1;
        nid.hIcon = LoadIcon(NULL, IDI_APPLICATION);
        lstrcpy(nid.szTip, TEXT("Meu ícone de notificação"));
        Shell_NotifyIcon(NIM_ADD, &nid);
        break;
    }
    case WM_DESTROY:
    {
        // Remover o ícone na barra de notificações
        NOTIFYICONDATA nid = { 0 };
        nid.cbSize = sizeof(NOTIFYICONDATA);
        nid.hWnd = hwnd;
        nid.uID = ID_TRAY_ICON;
        Shell_NotifyIcon(NIM_DELETE, &nid);

        // Encerrar a aplicação
        PostQuitMessage(0);
        break;
    }
    case WM_USER + 1:
    {
        // Lidar com eventos de clique no ícone
        if (lParam == WM_LBUTTONUP)
        {
            MessageBox(hwnd, TEXT("Ícone de notificação clicado!"), TEXT("Mensagem"), MB_ICONINFORMATION);
        }
        break;
    }
    default:
        return DefWindowProc(hwnd, uMsg, wParam, lParam);
    }
    return 0;
}

int main()
{
    // Registrar a classe da janela
    WNDCLASS wc = { 0 };
    wc.lpfnWndProc = WindowProc;
    wc.hInstance = GetModuleHandle(NULL);
    wc.lpszClassName = TEXT("TrayIconClass");
    RegisterClass(&wc);

    // Criar a janela (não visível)
    HWND hwnd = CreateWindowEx(0, TEXT("TrayIconClass"), TEXT(""), 0, 0, 0, 0, 0, NULL, NULL, GetModuleHandle(NULL), NULL);

    // Rodar o loop de mensagens
    MSG msg;
    while (GetMessage(&msg, NULL, 0, 0))
    {
        TranslateMessage(&msg);
        DispatchMessage(&msg);
    }

    return 0;
}
